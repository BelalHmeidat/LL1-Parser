module * Complex12Module;

const
    4PI = 3.14;

var
    radius:= real;
    area: real;

procedure CalculateArea
begin
    3area := PI * radius % radius;
end CalculateArea;

procede InputRadius; $
begin
    readreal(radius);
end InputRadius;

begin
    call InputRadius
    call CalculateArea.
    writereal(area);
end ComplexModule.
